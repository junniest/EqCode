\begin{eqcode}{advance}{p, v, m, dt}{\type{R}^2_{5,3}, \type{R}^2_{5,3}, \type{R}^1_5, \type{R}}{\type{R}^3}
    accs_{i,j}\ |\ 0 \leq i \leq 4 \land 0 \leq j \leq 4 =
        \begin{cases}
            \dfrac{(p_j-p_i) \cdot m_j}{\rho(p_i,p_j)^3} & j < i \lend
            0 & otherwise
        \end{cases} \lend
    accs_{i,j}\ |\ j > i = - accs_{j,i} \lend
    a_{i,j} = \sum \limits_k accs_{i,k,j} \lend
    v = v + a \cdot dt \lend
    p = p + v \cdot dt \lend
    \return {p, v} \lend
\end{eqcode}

\begin{eqcode}{energy}{p, v, m}{\type{R}^2_{5,3},
    \type{R}^2_{5,3}, \type{R}^1_5}{\type{R}}
    e \in \type{R} \lend
    e = \sum\limits_{i=0}^{4} \dfrac{m_i \cdot \sum\limits_{j} v_j^2}{2} \lend
    e = e - \sum\limits_{i=0, j=0}^{i<5,j<5}
    \begin{cases}
        \dfrac{m_i \cdot m_j}{\rho(p_i - p_j)} & i < j \lend
        0 & \text{otherwise} \lend
    \end{cases} \lend
    \return{e}
\end{eqcode}

\begin{eqcode}{offset\_momentum}{v, m}{\type{R}^2_{5,3},
    \type{R}^1_5}{\type{R}^2_{5,3}}
    v_{0,j}\ |\ 0 \leq j \leq 2 = \dfrac{-\sum\limits_{i} v_{i,j}
    \cdot m_i}{4 \cdot \pi^2} \lend
    \return{v} \lend
\end{eqcode}
